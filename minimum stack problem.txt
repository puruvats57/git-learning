#include<stdio.h>      //minimum stack problem   if doubt on meaning of  minimum stack search on google 
int main()
{
    int a[10]={4,3,5,1},i,s[10],top=0,m[10],j=1,top1,min;
    m[0]=a[0];
    min=a[0];


    for(i=0;i<4;i++)
    {
        s[top]=a[i];


        if(s[top]<min)
        {
            min=s[top];
            m[j]=s[top];
            j++;
        }
        top++;

    }
    --j;
    for(i=top-1;i>=0;i--)
    {
        if(s[i]==m[j])
        {
            printf("for %d min is %d\n",s[i],m[j]);
            j--;

        }
        else
            printf("for %d min is%d\n",s[i],m[j]);
    }






}
